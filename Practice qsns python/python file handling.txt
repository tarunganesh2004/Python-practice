questions :
1.IDs without S- prefix must be converted (e.g., 101 â†’ S-101).
2.Only FirstName,LastName are separated from full name.
3.City and extra fields are dropped.
4.Output strictly follows comma-separated format.

solution:

with open("input.txt",'r') as f:
    lines = f.readlines()
    data = [line.strip().split(",") for line in lines[0:]]
    for row in data:
        a = row[0].split("-")
        student_id= row[0]
        
        if(len(a) != 2):
            student_id = 'S' + '-' + row[0]
    
        fullname = row[1].split(" ")
        branch = row[2]
        semester = row[3]
        cgpa = row[5]
        row[0] = student_id
        row[1] = fullname[0][0].lower() + fullname[0][1:]
        row[2] = fullname[1]
        row[3] = branch
        row[4] = semester
        row[5] = cgpa
        
header = ["StudentID","FirstName","LastName","Branch","Semester","CGPA"]
with open("newinput.txt",'w') as f:
    f.write(",".join(header) + "\n")
    for row in data:
        f.write(",".join(row[:6]) + "\n")

with open("input.txt",'r') as f:
    lines = f.readlines()
    data = [line.strip().split(",") for line in lines[0:]]
    for row in data:
        student_id = row[0]
        a = len(row[0])
        if(a == 4 and row[0][0] == 'S'):
            student_id = 'S' + '-' + row[0][1:]
        elif(a == 3):
            student_id = 'S' + '-' + row[0][0:]
        else:
            student_id = 'S' + '-' + row[0][1:]

        fullname = row[1].strip().split(" ")
        firstname = fullname[0]
        lastname = fullname[1]

        branch = row[2].strip()
        semester = row[3]
        CGPA = row[5]

        row[0] = student_id
        row[1] = firstname
        row[2] = lastname
        row[3] = branch
        row[4] = semester
        row[5] = CGPA
